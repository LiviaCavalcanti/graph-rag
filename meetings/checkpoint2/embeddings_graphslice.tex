\section{Joern Parsing}
% ══════════════════════════════════════════════════════════════════════

\begin{frame}{Joern Parsing Issues: The Problem}
\begin{columns}[T]
\begin{column}{0.52\textwidth}

  \textbf{The symptom: \texttt{CASTProblemDeclaration}}\\[2pt]
  Joern's XML export labels nodes with
  \texttt{PARSER\_TYPE\_NAME = CASTProblemDeclaration} when it cannot
  resolve a type or function definition.  The entire construct
  collapses into a \textbf{single flat text node} — all internal
  structure (AST children, CFG, PDG edges) is lost.

  \vspace{0.6em}
  \textbf{Root causes in the dataset}\\[2pt]
  \begin{itemize}\setlength{\itemsep}{1pt}
    \item \textbf{Missing context} — type or function defined in a header
          not in the snippet
          (e.g.\ \texttt{vector}, \texttt{NotifyModelLoaded})
    \item \textbf{Modern C++ dialect} — \texttt{using type = ...},
          \texttt{auto\&\&}, range-based \texttt{for} unsupported
          by Joern's default C frontend
    \item \textbf{Unresolvable macros} — kernel/driver macros that expand
          only with the full build system
  \end{itemize}

\end{column}
\begin{column}{0.45\textwidth}

  \textbf{Structural impact on embeddings}\\[2pt]
  All code inside becomes a single text label.
  Structural embedders (GIN, WL, NetLSD) see a \textbf{degree-1 stub}
  where there should be a rich subgraph.

  \vspace{0.5em}
  \centering
  \begin{tikzpicture}[
    nd/.style={rectangle,rounded corners=2pt,draw=miss!60,fill=miss!8,
               font=\scriptsize\sffamily,minimum width=4.5cm,
               minimum height=0.5cm,align=left},
    gd/.style={rectangle,rounded corners=2pt,draw=hit!60,fill=hit!8,
               font=\scriptsize\sffamily,minimum width=4.5cm,
               minimum height=0.5cm,align=left}]
    \node[nd] (bad) at (0,2.2)
      {\textcolor{miss!70!black}{\textbf{1 node} — flat text blob}};
    \node[gd] (good) at (0,0.6)
      {\textcolor{hit!70!black}{\textbf{$N$ nodes} — full AST/CFG/PDG}};
    \draw[->,miss!50,thick] (0,2.9) node[above,font=\scriptsize\sffamily]
      {\texttt{CASTProblemDeclaration}} -- (bad);
    \draw[->,hit!50,thick] (0,1.3) node[above,font=\scriptsize\sffamily]
      {Resolved parse} -- (good);
    \node[font=\scriptsize\sffamily\bfseries,c3] at (0,-0.1)
      {Same source code, different structural info};
  \end{tikzpicture}

  \vspace{0.4em}
  {\scriptsize\color{black!55}
  Especially damaging for graph slicing:
  if the patch function is one opaque node, the diff
  cannot distinguish added/removed statements inside it.}

\end{column}
\end{columns}
\end{frame}

\begin{frame}{Joern Parsing Issues: Mitigation}
\begin{columns}[T]
\begin{column}{0.50\textwidth}

  \textbf{Before — standard Joern (C frontend)}\\[2pt]
  Example: \texttt{NotifyModelLoaded} function

  \vspace{0.3em}
  {\scriptsize\ttfamily\color{miss!70!black}
  \begin{tabular}{ll}
    node\_type & UNKNOWN \\
    PARSER\_TYPE\_NAME & \textbf{CASTProblemDeclaration} \\
    CODE & NotifyModelLoaded(model, \\
          & \quad type) \{ ... all body ... \}
  \end{tabular}}

  \vspace{0.3em}
  {\scriptsize\color{black!50}
  $\Rightarrow$ Single node, no children, no edges inside.\\
  The entire function body is one text label.}

  \vspace{0.7em}
  \textbf{After — C++ frontend + supplementary context}\\[2pt]

  \vspace{0.3em}
  {\scriptsize\ttfamily\color{hit!70!black}
  \begin{tabular}{ll}
    node\_type & METHOD \\
    name & NotifyModelLoaded \\
    CODE & (model, type) \{ ... \}
  \end{tabular}}

  \vspace{0.3em}
  {\scriptsize\color{black!50}
  $\Rightarrow$ Full AST/CFG/PDG subgraph with typed parameter\\
  nodes, call edges, and data-flow edges.}

\end{column}
\begin{column}{0.47\textwidth}

  \textbf{Two-part mitigation}

  \vspace{0.4em}
  \begin{block}{1. Switch to C++ frontend}
    \scriptsize Joern's newer C++ parser handles \texttt{auto\&\&},
    \texttt{using} aliases, range-based \texttt{for}, and other
    modern dialect features that the default C frontend rejects.
  \end{block}

  \vspace{0.3em}
  \begin{block}{2. Supplementary context file}
    \scriptsize Each CVE in the dataset includes a \emph{supplementary code}
    file with missing definitions: headers, typedefs, and
    macro stubs.  Feeding it to Joern alongside the patch
    resolves most \texttt{CASTProblemDeclaration} nodes.
  \end{block}

  \vspace{0.3em}
  {\scriptsize\color{black!55}
  Together these eliminate the majority of opaque nodes.
  Remaining cases are complex kernel macros requiring
  the full build environment — handled by the semantic
  (CodeBERT) embedder which can still read their text.}

\end{column}
\end{columns}
\end{frame}

\begin{frame}{Graph Slicing: Visualization}
\begin{columns}[T]
\begin{column}{0.50\textwidth}

  \textbf{Problem with the old approach}\\[3pt]
  Joern assigns non-deterministic node IDs across runs.
  ID-based diff produced $G_{\text{vuln}} \approx G_{\text{before}}$ —
  all functions appeared structurally identical.

  \vspace{0.9em}
  \textbf{After the fix}\\[3pt]
  Fingerprint-based matching yields a compact $G_{\text{vuln}}$
  containing only the vulnerability-relevant substructure:
  \begin{itemize}
    \item \textcolor{miss}{\textbf{Removed}} — node in $G_{\text{before}}$ only
    \item \textcolor{hit}{\textbf{Added}} — node in $G_{\text{after}}$ only
    \item \textcolor{warn!90!black}{\textbf{Context}} — unchanged $k$-hop neighbours\\
          reachable via PDG/CDG/CFG edges
  \end{itemize}

  \vspace{0.6em}
  {\scriptsize\color{black!55}
  The induced subgraph preserves the original edge types
  (AST, CFG, CDG, PDG) so that structural embedders (GIN, WL)
  can capture the local topology of the change.}

\begin{frame}[fragile]{Graph Slicing: Algorithm}
\begin{columns}[T]
  \begin{column}{0.52\textwidth}

    \textbf{Pseudocode — vulnerability subgraph extraction}

    \vspace{0.3em}
    {\scriptsize
    \begin{tabular}{@{}r@{\;\;}l}
      1: & \textbf{Input:} CPG$_{\text{before}}$, CPG$_{\text{after}}$, hop budget $k$ \\
      2: & \textbf{Output:} $G_{\text{vuln}}$ \\[2pt]
      3: & \textbf{for each} node $v$ \textbf{in} CPG$_{\text{before}}$ $\cup$ CPG$_{\text{after}}$: \\
      4: & \quad $\text{fp}(v) \gets (\text{node\_type}(v),\;\text{CODE}(v))$ \\[2pt]
      5: & $F_b \gets \{ \text{fp}(v) \mid v \in \text{CPG}_{\text{before}} \}$ \\
      6: & $F_a \gets \{ \text{fp}(v) \mid v \in \text{CPG}_{\text{after}} \}$ \\[2pt]
      7: & $\textcolor{miss}{\text{Removed}} \gets F_b \setminus F_a$ \\
      8: & $\textcolor{hit}{\text{Added}} \gets F_a \setminus F_b$ \\
      9: & $\text{Changed} \gets \textcolor{miss}{\text{Removed}} \cup \textcolor{hit}{\text{Added}}$ \\[2pt]
    10: & $\textcolor{warn!85!black}{\text{Context}} \gets \varnothing$ \\
    11: & \textbf{for each} $v \in \text{Changed}$: \\
    12: & \quad $\textcolor{warn!85!black}{\text{Context}} \mathrel{{+}{=}}
            \text{Nbrs}_{k}(v,\;\text{PDG}{\cup}\text{CDG}{\cup}\text{CFG})$ \\[2pt]
    13: & $V_{\text{vuln}} \gets \text{Changed} \cup \textcolor{warn!85!black}{\text{Context}}$ \\
    14: & \textbf{return} $G_{\text{vuln}} \gets
            \text{Induced}(\text{CPG}_{\text{after}},\;V_{\text{vuln}})$
    \end{tabular}}

    \vspace{0.4em}
    {\scriptsize Currently $k = 1$; context nodes add structural
    anchors without flooding $G_{\text{vuln}}$ with unchanged code.}

  \end{column}
  
  \begin{column}{0.45\textwidth}

    \textbf{Key steps illustrated}

\begin{frame}{Graph Slicing: Rationale}
\begin{columns}[T]
\begin{column}{0.55\textwidth}

  \textbf{Design rationale}

  \vspace{0.3em}
  \begin{itemize}\setlength{\itemsep}{4pt}
    \item \textbf{Fingerprint matching} (l.\,4) — invariant to Joern's
          non-deterministic node IDs; two runs of the same file
          yield the same fingerprint set.
    \item \textbf{Symmetric set difference} (l.\,7--8) — captures both
          deletions and insertions, following the standard
          \emph{program differencing} formulation [1].
    \item \textbf{$k$-hop context expansion} (l.\,11--12) — preserves
          the data- and control-flow neighbourhood of every change,
          grounded in Weiser's \emph{program slicing} criterion [2].
    \item \textbf{Induced subgraph} (l.\,14) — retains all original
          CPG edge types (AST, CFG, CDG, PDG) between selected nodes,
          so structural embedders see the full local topology [3].
  \end{itemize}

\end{column}
\begin{column}{0.42\textwidth}

  \textbf{References}

  \vspace{0.3em}
  {\small\color{black!70}
  \begin{enumerate}\setlength{\itemsep}{6pt}
    \item Horwitz — \emph{Identifying the semantic and textual
          differences between two versions of a program}, PLDI '90.
    \item Weiser — \emph{Program Slicing}, IEEE TSE, 1984.
    \item Yamaguchi et al. — \emph{Modeling and Discovering
          Vulnerabilities with Code Property Graphs}, IEEE S\&P '14.
  \end{enumerate}}

\end{column}
\end{columns}
\end{frame}

\begin{frame}{Diff Annotation: Design}

  \textbf{What — four-tier node labelling}\\[3pt]
  After the set-diff identifies changed nodes (step 1--2 of the
  algorithm) and the BFS expands context (step 3), every node in
  $G_{\text{vuln}}$ receives a \emph{diff type} and a scalar
  \emph{relevance weight}:

  \vspace{0.4em}
  {\scriptsize
  \renewcommand{\arraystretch}{1.15}
  \begin{tabular}{llcl}
    \toprule
    \textbf{Label} & \textbf{Meaning} & $w$ & \textbf{Detection} \\
    \midrule
    \texttt{removed}      & Code deleted by the patch
      & 1.0 & $\text{fp} \in F_b \setminus F_a$ \\
    \texttt{fix\_adjacent} & Neighbour of an inserted fix
      & 0.8 & adj.\ to added node in $G_{\text{after}}$ \\
    \texttt{edge\_changed} & Endpoint of a changed edge
      & 0.6 & edge fp differs before/after \\
    \texttt{context}       & Unchanged; reached by BFS
      & 0.2 & $k$-hop flow-edge expansion \\
    \bottomrule
  \end{tabular}}

  \vspace{0.5em}
  Embedders that use these weights:
  \begin{itemize}\setlength{\itemsep}{1pt}
    {
    \item \textbf{RGCN} — scalar node feature + weighted global pooling
    \item \textbf{CodeBERT seq} — filter ($w > 0.3$) and sort by $w$
    \item \textbf{GIN, WL, NetLSD} — use the diff \emph{label} (not weight)
          as a categorical node feature}
  \end{itemize}
\end{frame}

\begin{frame}{Diff Annotation: Design}

  \textbf{Why — causal proximity to the vulnerability}\\[3pt]
  The weight hierarchy reflects \textbf{distance from the actual
  code change}, following the principle that statements closer
  to the patch site carry more discriminative information for
  retrieval:

  \vspace{0.3em}
  \begin{enumerate}\setlength{\itemsep}{3pt}
    {\small
    \item \textbf{Directly changed code} ($w = 1.0$) —
          the vulnerability root cause; highest signal [1].
    \item \textbf{Fix-adjacent} ($w = 0.8$) —
          where the developer inserted the fix; marks the
          ``attack surface boundary'' [2].
    \item \textbf{Edge-changed} ($w = 0.6$) —
          structural ripple: a control/data-flow edge was
          rewired, indicating indirect semantic impact.
    \item \textbf{Context} ($w = 0.2$) —
          unchanged but reachable; provides structural
          anchors so embedders see local topology, following
          Weiser's slicing criterion [3].}
  \end{enumerate}

  \vspace{0.3em}
  {\scriptsize\color{black!55}
  [1] Li et al. — \emph{VulDeePecker: A Deep Learning-Based System
  for Vulnerability Detection}, NDSS '18.\\[1pt]
  [2] Yamaguchi et al. — \emph{Modeling and Discovering
  Vulnerabilities with Code Property Graphs}, IEEE S\&P '14.\\[1pt]
  [3] Weiser — \emph{Program Slicing}, IEEE TSE, 1984.}

\end{frame}

  \vspace{0.3em}

    \item $G_{\text{before}}$ has \textbf{no diff labels} —

\section{Key Findings}
% ══════════════════════════════════════════════════════════════════════

\begin{frame}{Key Findings}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
  \begin{enumerate}
    \item[\faIcon{bug}] \textbf{ID-based diff was fatally flawed.}\\
      Joern IDs are non-deterministic across runs.
      All graphs were indistinguishable.\\
      \textit{Fix: \texttt{(type, CODE)} fingerprint matching.}

    \vspace{0.7em}
    \item[\faIcon{layer-group}] \textbf{Combined is clearly the best structural embedder.}\\
      SR@1 = 86.3\%, CWE recall = 51.9\%.
      Concatenating all three structural signals (NetLSD + WL + GIN)
      outperforms each individually.

    \vspace{0.7em}
    \item[\faIcon{compress-arrows-alt}] \textbf{NetLSD collapses on small graphs.}\\
      Effective dimension $\approx 1$; pairwise cosine $> 0.99$.
      Cannot be used as sole embedder on $G_{\text{vuln}}$ subgraphs.
  \end{enumerate}
\end{column}
\begin{column}{0.48\textwidth}
  \begin{enumerate}
    \item[\faIcon{code}] \textbf{Structural and semantic signals are complementary.}\\
      GIN (structural) and CodeBERT (semantic) achieve the same SR@1 (75.3\%)
      but differ on CWE recall (39.8\% vs.\ 41.2\%).
      Neither dominates; a hybrid is the natural next step.

    \vspace{0.7em}
    \item[\faIcon{project-diagram}] \textbf{Errors are semantically coherent.}\\
      90.5\% of top-1 retrievals stay within the correct CWE.
      Confusions occur between structurally similar patch patterns
      (e.g.\ null-check $\approx$ lock-guard).
      This justifies knowledge-graph-based re-ranking.

    \vspace{0.7em}
    \item[\faIcon{exclamation-triangle}] \textbf{Self-retrieval is a lower bound.}\\
      No held-out query split yet.
      True generalisation performance is unknown.
  \end{enumerate}
\end{column}
\end{columns}
\end{frame}

\section{Next Steps}
% ══════════════════════════════════════════════════════════════════════

\begin{frame}{Next Steps}
% \begin{columns}[T]
% \begin{column}{0.60\textwidth}

  \textbf{1. Hybrid structural + semantic embedding}\\[2pt]
  Build a \texttt{HybridEmbedder} that concatenates GIN (graph topology)
  with CodeBERT (code semantics) into a single 128-d vector.
  GIN captures \emph{what changed structurally}; CodeBERT captures
  \emph{what the code means}. The confusion matrix shows they fail
  in different places.

  \vspace{0.8em}
  \textbf{2. Semantic graph diff for $G_{\text{vuln}}$}\\[2pt]
  Current diff uses exact string match on \texttt{CODE} fields — it
  misses rewrites that change logic while keeping function signatures.
  New approach: embed each node's code with CodeBERT and flag nodes
  whose embedding distance exceeds a threshold. Detects logical
  changes invisible to string comparison.

  \vspace{0.8em}
  \textbf{3. Held-out evaluation set}\\[2pt]
  Self-retrieval is circular — the query is always in the index.
  Need a genuinely unseen query set: real-world bug reports or
  cross-version CVE matches with known ground-truth targets.
  This gives honest precision/recall numbers for the thesis.
